```js
const input = new eg.Axes.RotatePanInput("#area");

var axes = new eg.Axes({
	// property name('angle') could be anything you want (eg. x, y, z...)
	angle: {
		range: [-180, 180] // from -180deg to 180deg
	}
});

axes.connect("angle", input)
```
